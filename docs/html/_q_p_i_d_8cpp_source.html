298 /* [now private]
299  @brief Due to inertia, DC motors don't start to run at very low speeds, they need a "kick" to get going. After that they'll
300  continue running even at low speeds. This function will apply this "kick" to a given input power. It will increase power to QPID::kickThreshold,
301  if *all* of the following conditions are met:
302  (1) QPID::kickThreshold is different from 0, (2) abs(power) is lower than QPID::kickThreshold ;
303  (3) abs(speed) is lower than or equal to QPID::kickThresholdSpeed , and (4) power is different from zero
304  @brief Disable kick by setting QPID::kickThreshold to 0
305  @param power original PWM value (-maxPWM to maxPWM)
306  @returns QPID::kickThreshold (sign according to power) if kick conditions met, else returns power as it was
307 */
308 /**************************************************************************/
309 
310 double QPID::kicked(double power) {
311  if ((kickThreshold == 0) or
312  (abs(power) >= kickThreshold) or
313  (abs(currentSpeed) > kickThresholdSpeed) or
314  (power == 0)) {
315  return power; // leave unchanged
316  } else { // apply kick
317  Serial.print(" KICK! ");
318  return (power > 0) ? kickThreshold : (0 - kickThreshold);
319  }
320 }
