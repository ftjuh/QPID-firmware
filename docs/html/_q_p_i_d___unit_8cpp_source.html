326 /**************************************************************************/
327 /* [now private]
328  @brief Due to inertia, DC motors don't start to run at very low speeds, they need a "kick" to get going. After that they'll
329  continue running even at low speeds. This function will apply this "kick" to
330  a given input power. It will increase power to QPID_Unit::kickThreshold,
331  if *all* of the following conditions are met:
332  (1) QPID_Unit::kickThreshold is different from 0, (2) abs(power) is lower
333  than QPID_Unit::kickThreshold ;
334  (3) abs(speed) is lower than or equal to QPID_Unit::kickThresholdSpeed , and
335  (4) power is different from zero
336  @brief Disable kick by setting QPID_Unit::kickThreshold to 0
337  @param power original PWM value (-maxPWM to maxPWM)
338  @returns QPID_Unit::kickThreshold (sign according to power) if kick
339  conditions met, else returns power as it was
340 */
341 /**************************************************************************/
342 
343 double QPID_Unit::kicked(double power) {
344  if ((kickThreshold == 0) or
345  (abs(power) >= kickThreshold) or
346  (abs(currentSpeed) > kickThresholdSpeed) or
347  (power == 0)) {
348  return power; // leave unchanged
349  } else { // apply kick
350  log("QPID_Unit kick applied.");
351  return (power > 0) ? kickThreshold : (0 - kickThreshold);
352  }
353 }
