56 void QPID_MotorDriver_DRV8833::setPower(int16_t pow) {
57  if (abs(pow) <= alwaysOnPWM) { // pwm valid?
58  currentPWM = pow;
59  if (pow == 0) { // setPower(0) is not recommended, better use coast() or brake()
60  analogWrite(pinA, decayMode);
61  analogWrite(pinB, decayMode);
62  } else {
63  // fill in logic table, [1] and [2] are prefilled in init(). Using a logic table saves us a lot of if() statements
64  // *and* ensures equal execution time independent from decayMode and direction (not sure if this is relevant, but feels safer this way).
65  // note: slowDecay needs reversed PWM (cell 3), the datasheet doesn't mention this.
66  logicTable[0] = abs(pow); logicTable[3] = 255-abs(pow);
67  int8_t idx = (pow < 0) << 1 | decayMode; // compute table cell for pinA
68  analogWrite(pinA, logicTable[idx]);
69  analogWrite(pinB, logicTable[idx ^ 0x02]); // pinB uses table cell shifted by 2 (idx + 2 % 4)
70  //Serial.print(" "); Serial.print(idx); Serial.print("|"); Serial.print(idx ^ 0x02); Serial.print(" ");
